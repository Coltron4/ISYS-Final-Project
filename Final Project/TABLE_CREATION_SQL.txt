CREATE TABLE MAKS202.SOCIAL_USERS (
	USERNAME VARCHAR(20) NOT NULL PRIMARY KEY,
	PASS VARCHAR(20),
	REAL_NAME VARCHAR(50)
);

CREATE TABLE MAKS202.SOCIAL_FOLLOWINGS (
	USERNAME VARCHAR(20) NOT NULL REFERENCES SOCIAL_USERS(USERNAME),
	FOLLOWER_USERNAME VARCHAR(20) NOT NULL REFERENCES SOCIAL_USERS(USERNAME)
);

CREATE TABLE MAKS202.SOCIAL_POSTS (
	POSTER_USERNAME VARCHAR(20) NOT NULL REFERENCES SOCIAL_USERS(USERNAME),
	POST_TEXT VARCHAR(250),
	POST_TIME TIMESTAMP
);
